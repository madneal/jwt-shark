### 插入排序

```go
// input : 5, 2, 4, 6, 1,3
// output : 1, 2, 3, 4, 5, 6
func insert_sort(nums [6]int) [6]int {
  j := 1
  for j < len(nums) {
    key := nums[j]
    i := j - 1
    for i >= 0 && nums[i] > key {
      nums[i+1] = nums[i]
      i = i - 1
    }
    nums[i + 1] = key
    j = j + 1
  }
  return nums
}
```
